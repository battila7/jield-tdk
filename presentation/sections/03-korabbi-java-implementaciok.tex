\begin{lstlisting}[language=java, xleftmargin=15pt,

        numbersep=5pt, escapechar=!]
@Continuable
public Iterable<Integer> power(int number, int exponent) 
{
        int counter = 0;
        int result = 1;
        while (counter++ < exponent) {
                result = result * number;

                System.out.print("[" + result + "]");

                Yield.ret(result);
        }
        return Yield.done();
}
\end{lstlisting}
